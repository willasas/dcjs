5. 视频滤镜示例

点击下方按钮应用视频滤镜。

点击按钮开始测试
// 应用视频滤镜 DC.Media.getMediaStream({ video: true, audio: false }) .then(stream => { const video = document.getElementById('filter-video'); video.srcObject = stream; video.style.display = 'none'; const canvas = document.getElementById('filter-canvas'); const stopFilter = DC.Media.applyVideoFilter(video, canvas, (data) => { // 黑白滤镜 for (let i = 0; i < data.length; i += 4) { const gray = (data[i] + data[i + 1] + data[i + 2]) / 3; data[i] = gray; // 红色 data[i + 1] = gray; // 绿色 data[i + 2] = gray; // 蓝色 } }); // 存储停止函数以便后续调用 window.stopFilter = stopFilter; }) .catch(error => { console.error('获取视频流失败:', error); });

// 大文件上传示例 function testUploadWithBackpressure() { const fileInput = document.createElement('input'); fileInput.type = 'file'; fileInput.onchange = function() { const file = this.files[0]; if (!file) return; DC.Media.uploadWithBackpressure(file, '/upload', { chunkSize: 1024 * 1024, // 1MB chunks onProgress: (progress) => { console.log('上传进度:', progress.percentage); } }); }; fileInput.click(); } // 实时数据处理示例 function testProcessRealtimeData() { // 模拟实时数据流 function createMockStream() { let i = 0; return new ReadableStream({ start(controller) { const interval = setInterval(() => { if (i > 10) { clearInterval(interval); controller.close(); return; } const data = new Uint8Array(1024); controller.enqueue(data); i++; }, 100); } }); } const mockStream = createMockStream(); const stopProcessing = DC.Media.processRealtimeData(mockStream, async (data) => { console.log('处理数据块:', data.byteLength); }); } // 流转换示例 function testTransformStream() { // 模拟数据源 function createSourceStream() { let i = 0; return new ReadableStream({ start(controller) { const interval = setInterval(() => { if (i > 5) { clearInterval(interval); controller.close(); return; } controller.enqueue(i); i++; }, 200); } }); } const sourceStream = createSourceStream(); const transformFn = async (value) => value * 2; const transformedStream = DC.Media.transformStream(sourceStream, transformFn); const reader = transformedStream.getReader(); const process = async () => { while (true) { const { done, value } = await reader.read(); if (done) break; console.log('转换后的值:', value); } }; process(); } // 大文件下载示例 function testDownloadWithProgress() { DC.Media.downloadWithProgress('https://example.com/large-file.zip', { onProgress: (progress) => { console.log('下载进度:', progress.percentage); } }) .then(blob => { console.log('下载完成:', blob.size); }); }
