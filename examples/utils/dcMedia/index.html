5. 视频滤镜示例

点击下方按钮应用视频滤镜。

点击按钮开始测试
// 应用视频滤镜 DC.Media.getMediaStream({ video: true, audio: false }) .then(stream => { const video = document.getElementById('filter-video'); video.srcObject = stream; video.style.display = 'none'; const canvas = document.getElementById('filter-canvas'); const stopFilter = DC.Media.applyVideoFilter(video, canvas, (data) => { // 黑白滤镜 for (let i = 0; i < data.length; i += 4) { const gray = (data[i] + data[i + 1] + data[i + 2]) / 3; data[i] = gray; // 红色 data[i + 1] = gray; // 绿色 data[i + 2] = gray; // 蓝色 } }); // 存储停止函数以便后续调用 window.stopFilter = stopFilter; }) .catch(error => { console.error('获取视频流失败:', error); });

6. 媒体录制示例

点击下方按钮开始录制视频。

点击按钮开始测试
// 开始录制 let mediaRecorder; let recordedChunks = []; DC.Media.getMediaStream() .then(stream => { const video = document.getElementById('recording-video'); video.srcObject = stream; video.style.display = 'block'; mediaRecorder = DC.Media.recordStream(stream); }) .catch(error => { console.error('获取媒体流失败:', error); }); // 停止录制 mediaRecorder.then(blob => { const url = URL.createObjectURL(blob); const a = document.createElement('a'); a.href = url; a.download = 'recording.webm'; a.click(); });
