什么是 document.currentScript？

document.currentScript 是一个只读属性，返回当前正在执行的 <script> 元素。这对于需要知道自身脚本路径的脚本非常有用，尤其是在动态加载依赖资源时。

document.currentScript // 返回当前正在执行的脚本元素

使用场景

  • 动态加载与当前脚本相关的其他资源（CSS、JS等）
  • 获取脚本所在目录，加载相对路径的依赖
  • 在模块化系统中解析依赖路径
  • 创建可重用的组件库，自动加载所需资源

// 获取当前脚本路径 function getScriptPath() { const script = document.currentScript; if (!script) return null; return script.src.substring(0, script.src.lastIndexOf('/') + 1); } // 动态加载CSS function loadCSS(url) { return new Promise((resolve, reject) => { const link = document.createElement('link'); link.rel = 'stylesheet'; link.href = url; link.onload = () => resolve(link); link.onerror = () => reject(new Error(`Failed to load CSS: ${url}`)); // 安全地添加到head if (document.head) { document.head.appendChild(link); } else { reject(new Error('document.head is not available')); } }); } // 动态加载JavaScript function loadScript(url) { return new Promise((resolve, reject) => { const script = document.createElement('script'); script.src = url; script.onload = () => resolve(script); script.onerror = () => reject(new Error(`Failed to load script: ${url}`)); // 安全地添加到head if (document.head) { document.head.appendChild(script); } else { reject(new Error('document.head is not available')); } }); } // 加载依赖 async function loadDependencies() { const basePath = getScriptPath(); if (!basePath) { throw new Error('无法获取当前脚本路径'); } try { // 加载CSS依赖 await loadCSS(basePath + 'dependency.css'); log('CSS依赖加载成功'); // 加载JS依赖 await loadScript(basePath + 'dependency.js'); log('JS依赖加载成功'); return true; } catch (error) { log(`加载依赖失败: ${error.message}`, 'error'); return false; } }
